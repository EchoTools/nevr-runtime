  std::vector<ovrID> CustomSocialManager::GetPlaceholderFriends(ovrID user_id) {
    static const std::vector<ovrID> placeholder_friends = {
        0x1000000000000001ULL,  // Edit these
        0x1000000000000002ULL,  //
        0x1000000000000003ULL,  //
        0x1000000000000004ULL   //
    };
    return placeholder_friends;
  }

Change the 4 constants to whatever friend IDs you want returned.
